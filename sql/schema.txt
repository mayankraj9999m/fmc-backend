CREATE SCHEMA "public";
CREATE TYPE "admin_position" AS ENUM('Chief Warden', 'Hostel Warden', 'Associate Warden', 'Junior Assistant');
CREATE TYPE "complaint_status" AS ENUM('Initiated', 'Worker assigned', 'Resolved');
CREATE TABLE "admins" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"name" varchar(255) NOT NULL,
	"email" varchar(255) NOT NULL CONSTRAINT "admins_email_key" UNIQUE,
	"password_hash" varchar(255) NOT NULL,
	"phone_no" varchar(20),
	"photo" text,
	"position" admin_position NOT NULL,
	"hostel_name" varchar(100),
	"created_at" timestamp with time zone DEFAULT CURRENT_TIMESTAMP,
	"last_login" timestamp with time zone,
	"requires_password_change" boolean DEFAULT true
);
CREATE TABLE "complaints" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"complaint_no" serial CONSTRAINT "complaints_complaint_no_key" UNIQUE,
	"student_id" uuid,
	"department" varchar(100) NOT NULL,
	"sub_category" varchar(100),
	"description" varchar(300),
	"status" complaint_status DEFAULT 'Initiated',
	"complaint_image" text,
	"resolved_image" text,
	"is_escalated" boolean DEFAULT false,
	"worker_id" uuid,
	"rating" integer,
	"feedback" text,
	"lodged_at" timestamp with time zone DEFAULT CURRENT_TIMESTAMP,
	"assigned_at" timestamp with time zone,
	"resolved_at" timestamp with time zone,
	CONSTRAINT "complaints_rating_check" CHECK (((rating >= 1) AND (rating <= 5)))
);
CREATE TABLE "hostels" (
	"name" varchar(100) PRIMARY KEY,
	"capacity" integer,
	"created_at" timestamp with time zone DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE "students" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"google_id" varchar(255) CONSTRAINT "students_google_id_key" UNIQUE,
	"roll_no" varchar(50) CONSTRAINT "students_roll_no_key" UNIQUE,
	"name" varchar(255) NOT NULL,
	"email" varchar(255) NOT NULL CONSTRAINT "students_email_key" UNIQUE,
	"profile_picture" text,
	"hostel_name" varchar(100),
	"room_no" varchar(20),
	"floor_no" integer,
	"gender" varchar(20),
	"phone_no" varchar(20),
	"branch" varchar(100),
	"year_of_joining" integer,
	"is_onboarded" boolean DEFAULT false,
	"created_at" timestamp with time zone DEFAULT CURRENT_TIMESTAMP,
	"last_login" timestamp with time zone,
	"programme" varchar(50)
);
CREATE TABLE "workers" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"name" varchar(255) NOT NULL,
	"email" varchar(255) NOT NULL CONSTRAINT "workers_email_key" UNIQUE,
	"password_hash" varchar(255) NOT NULL,
	"phone_no" varchar(20),
	"gender" varchar(20),
	"photo" text,
	"hostel_name" varchar(100),
	"department" varchar(100) NOT NULL,
	"sub_work_category" varchar(100),
	"current_rating" numeric(3, 2) DEFAULT '0.00',
	"rating_count" integer DEFAULT 0,
	"created_at" timestamp with time zone DEFAULT CURRENT_TIMESTAMP,
	"last_login" timestamp with time zone
);
CREATE UNIQUE INDEX "admins_email_key" ON "admins" ("email");
CREATE UNIQUE INDEX "admins_pkey" ON "admins" ("id");
CREATE UNIQUE INDEX "complaints_complaint_no_key" ON "complaints" ("complaint_no");
CREATE UNIQUE INDEX "complaints_pkey" ON "complaints" ("id");
CREATE INDEX "idx_complaints_status" ON "complaints" ("status");
CREATE INDEX "idx_complaints_student" ON "complaints" ("student_id");
CREATE INDEX "idx_complaints_worker" ON "complaints" ("worker_id");
CREATE UNIQUE INDEX "hostels_pkey" ON "hostels" ("name");
CREATE INDEX "idx_students_google_id" ON "students" ("google_id");
CREATE UNIQUE INDEX "students_email_key" ON "students" ("email");
CREATE UNIQUE INDEX "students_google_id_key" ON "students" ("google_id");
CREATE UNIQUE INDEX "students_pkey" ON "students" ("id");
CREATE UNIQUE INDEX "students_roll_no_key" ON "students" ("roll_no");
CREATE UNIQUE INDEX "workers_email_key" ON "workers" ("email");
CREATE UNIQUE INDEX "workers_pkey" ON "workers" ("id");
ALTER TABLE "admins" ADD CONSTRAINT "fk_admins_hostel" FOREIGN KEY ("hostel_name") REFERENCES "hostels"("name") ON DELETE SET NULL ON UPDATE CASCADE;
ALTER TABLE "complaints" ADD CONSTRAINT "complaints_student_id_fkey" FOREIGN KEY ("student_id") REFERENCES "students"("id") ON DELETE CASCADE;
ALTER TABLE "complaints" ADD CONSTRAINT "complaints_worker_id_fkey" FOREIGN KEY ("worker_id") REFERENCES "workers"("id") ON DELETE SET NULL;
ALTER TABLE "students" ADD CONSTRAINT "fk_students_hostel" FOREIGN KEY ("hostel_name") REFERENCES "hostels"("name") ON DELETE SET NULL ON UPDATE CASCADE;
ALTER TABLE "workers" ADD CONSTRAINT "fk_workers_hostel" FOREIGN KEY ("hostel_name") REFERENCES "hostels"("name") ON DELETE SET NULL ON UPDATE CASCADE;
